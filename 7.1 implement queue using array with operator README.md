#include <stdio.h>
#define MAX_SIZE 100

int queue[MAX_SIZE];
int front = -1;
int rear = -1;

void enqueue(int num) {
    if (rear == MAX_SIZE - 1) {
        printf("Queue is full. Cannot enqueue.\n");
        
    } else if (front == -1 && rear == -1) {
        rear = front = 0;
        queue[rear] = num;
    } else {
        rear = rear + 1;
        queue[rear] = num;
    }

    printf("%d enqueued successfully.\n", num);
}

void dequeue() {
    if (front == -1 || rear == -1) {
        printf("Queue is empty. Cannot dequeue.\n");
        
    } else if (rear == front) {
        printf("%d dequeued successfully.\n", queue[front]);
        rear = front = -1;
    } else {
        printf("%d dequeued successfully.\n", queue[front]);
        queue[front++];
    }
}

void display() {
    if (front == -1 && rear==-1) {
        printf("Queue is empty.\n");
       
    }
    printf("Queue elements: ");
    for (int i = front; i <= rear; i++) {
        printf("%d ", queue[i]);
    }
    printf("\n");
}

int main() {
    enqueue(10);
    enqueue(20);
    enqueue(30);
    display();
    dequeue();
    display();
    return 0;
}
